PREFIX dc: <http://purl.org/dc/elements/1.1/>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT DISTINCT ?CulturalProperty ?title ?creator ?spatial
WHERE {
  {
	?CulturalProperty a arco:HistoricOrArtisticProperty;
                  	dc:title ?title .
	FILTER(REGEX(?title, "paesaggio. Grizzana", "i")) .
  }
  UNION
  {
	?CulturalProperty a arco:CulturalProperty ;
                  	dc:creator ?creator .
	?creator foaf:name "Giorgio Morandi" .
  }
  UNION
  {
	?CulturalProperty a arco:CulturalProperty ;
                  	dct:spatial ?spatial .
	FILTER(REGEX(?spatial, "Grizzana", "i")) .
  }
  UNION
  {
	?CulturalProperty a arco:CulturalProperty ;
                  	dct:description ?description .
	FILTER(REGEX(?description, "rolling hills of Grizzana", "i")) .
  }
}
