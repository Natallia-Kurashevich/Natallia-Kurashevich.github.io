PREFIX rdf: 
PREFIX rdfs: 
PREFIX cpv: 
PREFIX a-cd: 

SELECT DISTINCT ?artist ?name
WHERE {
    ?artist rdf:type cpv:Person ;
            rdf:type l0:Agent ;
            rdfs:label ?name ;
            a-cd:hasProfession ?profession ;
            cpv:hasBirthPlace ?birthPlace.
   ?profession l0:name "pittore".
   ?birthPlace l0:name "Bologna".
}
ORDER BY ASC(?name)
